async function loadMessages() { const { data, error } = await supabaseClient .from("posts") .select("*") .order("created_at", { ascending: true }); if (error) { console.error(error); // Log if Supabase fails return; } const list = document.getElementById("messages"); list.innerHTML = ""; if (!data) return; data.forEach(row => { const li = document.createElement("li"); li.innerHTML = `${row.username}: ${row.message}`; list.appendChild(li); }); }
